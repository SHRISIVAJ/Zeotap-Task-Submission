# Import necessary libraries
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from sklearn.preprocessing import StandardScaler
from sklearn.cluster import KMeans
from sklearn.metrics import davies_bouldin_score

# Load the datasets
customers = pd.read_csv('Customers.csv')
products = pd.read_csv('Products.csv')
transactions = pd.read_csv('Transactions.csv')

# Preprocess the data
# Merge transactions with customers to get customer profiles
merged_data = transactions.merge(customers, on='CustomerID', how='left')

# Create a feature set for each customer
customer_features = merged_data.groupby('CustomerID').agg({
    'TotalValue': 'sum',
    'Quantity': 'sum',
    'ProductID': lambda x: x.nunique(),  # Unique products purchased
    'Region': 'first'  # Keep the region for reference
}).reset_index()

# Standardize the features for clustering
scaler = StandardScaler()
scaled_features = scaler.fit_transform(customer_features[['TotalValue', 'Quantity', 'ProductID']])

# Determine the optimal number of clusters using the Elbow Method
inertia = []
silhouette_scores = []
db_indices = []

for n_clusters in range(2, 11):
    kmeans = KMeans(n_clusters=n_clusters, random_state=42)
    kmeans.fit(scaled_features)
    
    inertia.append(kmeans.inertia_)
    db_index = davies_bouldin_score(scaled_features, kmeans.labels_)
    db_indices.append(db_index)

# Plot the Elbow Method results
plt.figure(figsize=(12, 6))
plt.subplot(1, 2, 1)
plt.plot(range(2, 11), inertia, marker='o')
plt.title('Elbow Method for Optimal k')
plt.xlabel('Number of Clusters')
plt.ylabel('Inertia')

plt.subplot(1, 2, 2)
plt.plot(range(2, 11), db_indices, marker='o')
plt.title('Davies-Bouldin Index for Optimal k')
plt.xlabel('Number of Clusters')
plt.ylabel('DB Index')

plt.show()

# Choose the optimal number of clusters (e.g., based on elbow method or lowest DB index)
optimal_clusters = 4  # You can change this based on your analysis

# Fit KMeans with the optimal number of clusters
kmeans_final = KMeans(n_clusters=optimal_clusters, random_state=42)
customer_features['Cluster'] = kmeans_final.fit_predict(scaled_features)

# Calculate clustering metrics
db_index_final = davies_bouldin_score(scaled_features, customer_features['Cluster'])

# Visualize clusters using a scatter plot (using first two features for simplicity)
plt.figure(figsize=(10, 6))
sns.scatterplot(x=customer_features['TotalValue'], y=customer_features['Quantity'], hue=customer_features['Cluster'], palette='viridis', s=100)
plt.title(f'Customer Segmentation with {optimal_clusters} Clusters')
plt.xlabel('Total Value (USD)')
plt.ylabel('Quantity Purchased')
plt.legend(title='Cluster')
plt.show()

# Output clustering results and metrics
print(f'Number of clusters formed: {optimal_clusters}')
print(f'Davies-Bouldin Index: {db_index_final}')
